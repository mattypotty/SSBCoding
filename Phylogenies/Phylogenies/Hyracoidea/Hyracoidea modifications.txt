#One of the trees of the family from the posterior distribution (only one family in the order)
Hyracoidea<-read.tree(text="((Heterohyrax_brucei:1,Procavia_capensis:1):1,Dendrohyrax_dorsalis:1);")
PHYLO<-Hyracoidea

#Place missing species of Dendrohyrax based on Nature 446, 507-512 (2007)
TR<-read.tree(text="((Dendrohyrax_arboreus:1, Dendrohyrax_validus:1):1, Dendrohyrax_dorsalis:1):1;")
PHYLO<<-bind.tree(PHYLO, TR ,  getNode(phylo4(PHYLO),"Dendrohyrax_dorsalis"))
 
#Simulate potential branch lengths wihtin the family based on MRCA of family equal to 6.07 MYA following Meredith et al (2011)
BD.gen(6.07,0.904)
PHYLO->Hyracoidea


################################################################################################################################################################
##########################################################Note that the order is identical to the family########################################################
################################################################################################################################################################

write.tree(Hyracoidea, "Hyracoidea.nex")
