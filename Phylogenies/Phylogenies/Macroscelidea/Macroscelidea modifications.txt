#One of the trees of the family from the posterior distribution. Only one family in the order
Macroscelidea<-read.tree(text="(Rhynchocyon_petersi:1,((((Elephantulus_rufescens:1,Elephantulus_brachyrhynchus:1):1,(Elephantulus_rupestris:1,Elephantulus_intufi:1):1):1,(Elephantulus_myurus:1,(Elephantulus_pilicaudus:1,Elephantulus_edwardii:1):1):1):1,(Macroscelides_proboscideus:1,(Petrodromus_tetradactylus:1,Elephantulus_rozeti:1):1):1):1);")
PHYLO<-Macroscelidea
#One of the trees for the genus Rhynchocyon from the posterior distribution
Rhynchocyon<-read.tree(text="((Rhynchocyon_petersi:1,Rhynchocyon_cirnei:1):1,Rhynchocyon_chrysopygus:1);")
MONOP.GROUP<-c("Rhynchocyon_chrysopygus", "Rhynchocyon_cirnei", "Rhynchocyon_petersi")
#Merging the family tree with the tree for Rhynchocyon
Replace.groups(Rhynchocyon, MONOP.GROUP)
#One of the trees for the genus Elephantulus from the posterior distribution
Elephantulus<-read.tree(text="((((Elephantulus_fuscipes:1,Elephantulus_fuscus:1):1,Elephantulus_revoili:1):1,((Elephantulus_myurus:1,(Elephantulus_edwardii:1,Elephantulus_pilicaudus:1):1):1,((Elephantulus_intufi:1,Elephantulus_rupestris:1):1,(Elephantulus_rufescens:1,Elephantulus_brachyrhynchus:1):1):1):1):1,Elephantulus_rozeti:1);")
MONOP.GROUP<-c("Elephantulus_brachyrhynchus", "Elephantulus_edwardii", "Elephantulus_fuscipes", "Elephantulus_fuscus", "Elephantulus_intufi", "Elephantulus_myurus", "Elephantulus_pilicaudus", "Elephantulus_revoili", "Elephantulus_rozeti", "Elephantulus_rufescens", "Elephantulus_rupestris")
#Merging the family tree with the tree for Elephantulus 
Replace.groups(Elephantulus, MONOP.GROUP)

#Place Rhynchocyon_udzungwensis based on taxonomy
SisterOrSing1(c("Rhynchocyon_chrysopygus", "Rhynchocyon_cirnei", "Rhynchocyon_petersi"), "Rhynchocyon_udzungwensis")

#Simulate potential branch lengths wihtin the family based on MRCA of family equal to 49.14 MYA following Meredith et al (2011)
BD.gen(49.14,0.904)
PHYLO->Macroscelidea

################################################################################################################################################################
##########################################################Note that the order is identical to the family########################################################
################################################################################################################################################################

write.tree(Macroscelidea, "Macroscelidea.nex")
