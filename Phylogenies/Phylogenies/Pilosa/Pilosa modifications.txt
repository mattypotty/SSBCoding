###########################################################Bradypodidae#########################################################################################
################################################################################################################################################################

#One of the trees of the family from the posterior distribution
Bradypodidae<-read.tree(text="(Bradypus_torquatus:1,(Bradypus_tridactylus:1,Bradypus_variegatus:1):1);")
PHYLO<-Bradypodidae

#Place Bradypus_pygmaeus based on former conspecificy
Sister1("Bradypus_variegatus", "Bradypus_pygmaeus")

#Simulate potential branch lengths wihtin the family based on a divergence time with its closest relatives of 22.2 MYA following Meredith et al (2011)
#A node called Rootnode_Bradypodidae is temporarily added but will be deleted when the tree is combined with the remaining families in the order
PHYLO<-bind.tree(read.tree(text="(ROOT_1:1, Rootnode_Bradypodidae:1);"), PHYLO,1)
BD.gen(22.2,0.904)
PHYLO->Bradypodidae

###########################################################The interfamily relationship#########################################################################
################################################################################################################################################################

#The relationsnip between the families in the order
Pilosa<-read.tree(text="((Cyclopes_didactylus:38.65, Myrmecophagidae:0):17.73,(Bradypodidae:0, Megalonychidae:0):34.18);")

#Adding the families one by one
Pilosa<-bind.tree(Pilosa, Bradypodidae, getNode(phylo4(Pilosa),"Bradypodidae")); Pilosa<-drop.tip(Pilosa, "Rootnode_Bradypodidae")
Pilosa<-bind.tree(Pilosa, Megalonychidae, getNode(phylo4(Pilosa),"Megalonychidae")); Pilosa<-drop.tip(Pilosa, "Rootnode_Megalonychidae")
Pilosa<-bind.tree(Pilosa, Myrmecophagidae, getNode(phylo4(Pilosa),"Myrmecophagidae")); Pilosa<-drop.tip(Pilosa, "Rootnode_Myrmecophagidae")

write.tree(Pilosa, "Pilosa.nex")
